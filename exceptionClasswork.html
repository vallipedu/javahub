ExceptionDemo.java 

import java.util.Scanner;

/**
 * This is a test class to demonstrate exception handling in Java.
 * It reads two integers from the user and prints their division.
 * It also includes a try-catch block to handle exceptions and a finally block
 * to ensure that resources are properly closed.
 * The program handles ArithmeticException for division by zero and
 * InputMismatchException for invalid input.
 * * @author HariShankar Mutchakala
 * Date: 2025-09-23
 * Version: 1.0
 * */
public class ExceptionDemo {

    public static void main(String a[]) {
        System.out.println("Hello World");
        // read two numbers from keyboard and print their division
        Scanner sc = new Scanner(System.in);

        System.out.println("Enter two numbers:");

        try {
            int num1 = sc.nextInt();
            System.out.println("Enter second number:");
            int num2 = sc.nextInt();
            int result = num1 / num2;
            System.out.println("The division of " + num1 + " and " + num2 + " is: " + result);
        } catch (ArithmeticException e) {
            // exception handler for division by zero goes here
            System.out.println("Error: Division by zero is not allowed.");
            System.out.println("Exception details: " + e.getMessage());
        } finally {
            // safety net
            sc.close();
            System.out.println("Finally block executed");
        }
        System.out.println("After catch block");
        System.out.println("End of the program");

    }
}
MultipleCatch.java 

public class MultipleCatch {

    public static void main(String[] args) {
        try {
            int[] numbers = { 1, 2, 3 };
            System.out.println(numbers[5]);
            int result = 10 / 0;
            System.out.println("Result: " + result);
        } catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("Array index is out of bounds");
            
        } catch (ArithmeticException e) {
            System.out.println("Cannot divide by zero");
            
        } catch (Exception e) {
            System.out.println("An unexpected error occurred");
          
        }
        System.out.println("End of program");
    }
}
MultipleTryCatchBlocks.java 

/**
 * this class demonstrates multiple try-catch blocks.
 * It handles ArrayIndexOutOfBoundsException and ArithmeticException separately.
 * Each try block is followed by its own catch block.
 * This allows for more granular error handling.
 * The program continues execution after handling each exception.
 * Author: HariShankar Mutchakala
 * Date: 2025-09-23
 * Version: 1.0
 * */
public class MultipleTryCatchBlocks {

    public static void main(String[] args) {
        try {
            int[] numbers = { 1, 2, 3 };
            System.out.println(numbers[5]);
        } catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("Array index is out of bounds");
        }

        try {
            int result = 10 / 0;
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            System.out.println("Cannot divide by zero");
        }

        System.out.println("End of program");
    }
}
ThrowsDemo.java 

/**
 * * This class demonstrates throw and throws in Java.
 * It includes methods that throw exceptions and a main method that handles them.
 * Author: HariShankar Mutchakalar
 * Date: 2025-09-23
 * Version: 1.0
 * */
public class ThrowsDemo {
    public static void main(String[] args) {
        try {
            methodA();
        } catch (Exception e) {
            System.out.println("Exception caught in main: " + e.getMessage());
        }
        System.out.println("End of program");    
    }
    static void methodA() throws Exception {
        methodB();
    }

    static void methodB() throws Exception {
        throw new Exception("Exception thrown in methodB");
    }
}
BlanketCatchDemo.java 

/**
 * * * Demonstrates a blanket catch block that catches all exceptions.
 * It handles both ArrayIndexOutOfBoundsException and ArithmeticException
 * in a single catch block.
 * The program continues execution after handling the exception.
 * Author: HariShankar Mutchakala
 * Date: 2025-09-23
 * Version: 1.0
 */
public class BlanketCatchDemo {

    public static void main(String[] args) {
        try {
            int[] numbers = { 1, 2, 3 };

UserDefinedBusinessExceptionDemo.java 

// UserDefinedBusinessExceptionDemo.java
// Custom exception class
//this exception is thrown when a withdrawal amount exceeds the available balance     

class InSufficientFundsException extends Exception {

    public InSufficientFundsException(String message) {
        super(message);
    }
}

/**
 * this class demonstrates user-defined business exceptions.
 * It demonstrates how to create custom business exceptions.
 * In this example, we create a custom exception called
 * InSufficientFundsException.
 * It is thrown when a withdrawal amount exceeds the available balance.
 * The main method calls the withdraw method, which throws this exception if the
 * balance is insufficient.
 * The main method catches the exception and prints the error message.
 * The main method also demonstrates how to handle the exception using a
 * try-catch block.
 * The main method demonstrates that the exception is caught and handled
 * appropriately.
 * * * @author HariShankar Mutchakala
 * Date: 2025-09-23
 * * Version: 1.0
 * */

public class UserDefinedBusinessExceptionDemo {
    public static void main(String[] args) {
        try {
            double amount = 5000;
            double balance = 2000;
            withdraw(balance, amount);
        } catch (InSufficientFundsException e) {
            System.out.println("Exception caught in main: " + e.getMessage());
        }
    }

    static void withdraw(double balance, double amount) throws InSufficientFundsException {
        if (balance - amount < 0) {
            throw new InSufficientFundsException("Insufficient funds");
        }
        balance -= amount; // update balance
        System.out.println("Withdrawn: " + amount);
        System.out.println("Remaining balance: " + balance);
    }
}
